001package com.github.mk23.jmxproxy.conf;
002
003import com.fasterxml.jackson.annotation.JsonProperty;
004
005import io.dropwizard.Configuration;
006
007import javax.validation.Valid;
008
009/**
010 * <p>Main top-level configuration.</p>
011 *
012 * Maintains the top-level application configuration object. The
013 * application initializer creates and marshals the object from
014 * JSON or Yaml config file.
015 *
016 * @since   2016-01-28
017 * @author  mk23
018 * @version 3.2.1
019 */
020public class MainConfig extends Configuration {
021    /**
022     * Application service config.
023     */
024    @Valid
025    @JsonProperty("jmxproxy")
026    private AppConfig appConfig = new AppConfig();
027
028    /**
029     * <p>Getter for appConfig.</p>
030     *
031     * Application service config.
032     *
033     * @return Application subsection from the full configuration file.
034     */
035    public final AppConfig getAppConfig() {
036        return appConfig;
037    }
038    /**
039     * <p>Setter for appConfig.</p>
040     *
041     * @param appConfig Application subsection from the full configuration file.
042     */
043    public final void setAppConfig(final AppConfig appConfig) {
044        this.appConfig = appConfig;
045    }
046}
