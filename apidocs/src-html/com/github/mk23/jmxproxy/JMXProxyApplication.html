001package com.github.mk23.jmxproxy;
002
003import com.github.mk23.jmxproxy.conf.MainConfig;
004import com.github.mk23.jmxproxy.jmx.ConnectionManager;
005
006import io.dropwizard.Application;
007import io.dropwizard.assets.AssetsBundle;
008import io.dropwizard.server.AbstractServerFactory;
009import io.dropwizard.setup.Bootstrap;
010import io.dropwizard.setup.Environment;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * <p>JMXProxy main application.</p>
017 *
018 * Configures application from {@link MainConfig}, initializes, and starts the service.
019 *
020 * @see <a href="http://dropwizard.github.io/dropwizard/0.9.3/dropwizard-core/apidocs/io/dropwizard/Application.html">io.dropwizard.Application</a>
021 * @see <a href="http://dropwizard.github.io/dropwizard/0.9.3/dropwizard-core/apidocs/io/dropwizard/setup/Bootstrap.html">io.dropwizard.setup.Bootstrap</a>
022 * @see <a href="http://dropwizard.github.io/dropwizard/0.9.3/dropwizard-core/apidocs/io/dropwizard/setup/Environment.html">io.dropwizard.setup.Environment</a>
023 *
024 * @since   2015-05-11
025 * @author  mk23
026 * @version 3.2.0
027 */
028public class JMXProxyApplication extends Application<MainConfig> {
029    private static final Logger LOG = LoggerFactory.getLogger(JMXProxyApplication.class);
030
031    /**
032     * <p>main application entrypoint.</p>
033     *
034     * Starts main application.
035     *
036     * @param args an array of {@link String} command-line parameters.
037     *
038     * @throws Exception if initialization fails.
039     */
040    public static void main(final String[] args) throws Exception {
041        LOG.info("starting jmxproxy service");
042        new JMXProxyApplication().run(args);
043    }
044
045    /** {@inheritDoc} */
046    @Override
047    public final String getName() {
048        return "jmxproxy";
049    }
050
051    /** {@inheritDoc} */
052    @Override
053    public final void initialize(final Bootstrap<MainConfig> bootstrap) {
054        bootstrap.addBundle(new AssetsBundle("/assets/", "/", "index.html"));
055    }
056
057    /** {@inheritDoc} */
058    @Override
059    public final void run(final MainConfig configuration, final Environment environment) {
060        final ConnectionManager manager = new ConnectionManager(configuration.getAppConfig());
061        final JMXProxyResource resource = new JMXProxyResource(manager);
062        final JMXProxyHealthCheck healthCheck = new JMXProxyHealthCheck(manager);
063
064        ((AbstractServerFactory) configuration.getServerFactory()).setJerseyRootPath("/jmxproxy/*");
065
066        environment.lifecycle().manage(manager);
067        environment.jersey().register(resource);
068        environment.healthChecks().register("manager", healthCheck);
069    }
070}
