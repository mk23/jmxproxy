472    private Response getJMXHost(
473        final String hostName,
474        final int port,
475        final boolean full,
476        final int limit,
477        final ConnectionCredentials auth
478    ) {
479        Host host = manager.getHost(hostName + ":" + port, auth);
480        return Response.ok(full ? host.setLimit(limit) : host.getMBeans()).build();
481    }
482
483    private Response getJMXHost(
484        final String hostName,
485        final int port,
486        final String mbeanName,
487        final boolean full,
488        final int limit,
489        final ConnectionCredentials auth
490    ) {
491        MBean mbean = manager.getHost(hostName + ":" + port, auth).getMBean(mbeanName);
492        if (mbean == null) {
493            return Response.status(Response.Status.NOT_FOUND).build();
494        }
495
496        return Response.ok(full ? mbean.setLimit(limit) : mbean.getAttributes()).build();
497    }
498
499    private Response getJMXHost(
500        final String hostName,
501        final int port,
502        final String mbeanName,
503        final String attribute,
504        final int limit,
505        final ConnectionCredentials auth
506    ) {
507        MBean mbean = manager.getHost(hostName + ":" + port, auth).getMBean(mbeanName);
508        if (mbean == null || !mbean.hasAttribute(attribute)) {
509            return Response.status(Response.Status.NOT_FOUND).build();
510        }
511
512        if (limit < 0) {
513            return Response.ok(mbean.getAttribute(attribute)).build();
514        } else {
515            return Response.ok(mbean.getAttributes(attribute, limit)).build();
516        }
517    }
