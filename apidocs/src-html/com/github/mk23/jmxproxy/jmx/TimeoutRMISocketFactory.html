001package com.github.mk23.jmxproxy.jmx;
002
003import java.io.IOException;
004
005import java.net.InetSocketAddress;
006import java.net.Socket;
007import java.net.ServerSocket;
008
009import java.rmi.server.RMISocketFactory;
010
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014/**
015 * <p>Remote Method Invocation socket factory with connection timeout.</p>
016 *
017 * Extends RMISocketFactory to return a socket that implements a connection timeout,
018 * preventing black-holed backend agents from piling up connections in the server.
019 *
020 * @see <a href="https://docs.oracle.com/javase/7/docs/api/java/net/Socket.html">java.net.Socket</a>
021 * @see <a href="https://docs.oracle.com/javase/7/docs/api/java/rmi/server/RMISocketFactory.html">java.rmi.server.RMISocketFactory</a>
022 *
023 * @since   2017-01-20
024 * @author  mk23
025 * @version 3.3.6
026 */
027public class TimeoutRMISocketFactory extends RMISocketFactory {
028    private static final Logger LOG = LoggerFactory.getLogger(TimeoutRMISocketFactory.class);
029
030    private int timeout;
031
032    /**
033     * <p>Default constructor.</p>
034     *
035     * Initializes the timeout to the specified value in milliseconds.
036     *
037     * @param timeout milliseconds to pass to Socket's connect() method.
038     */
039    public TimeoutRMISocketFactory(final int timeout) {
040        super();
041
042        this.timeout = timeout;
043    }
044
045    /**
046     * <p>Setter for timeout.</p>
047     *
048     * Resets the timeout to the specified milliseconds value.
049     *
050     * @param timeout milliseconds to pass to Socket's connect() method.
051     *
052     * @return Modified TimeoutRMISocketFactory for setter chaining.
053     */
054    public final TimeoutRMISocketFactory setTimeout(final int timeout) {
055        this.timeout = timeout;
056        return this;
057    }
058
059    /** {@inheritDoc} */
060    @Override
061    public final Socket createSocket(final String host, final int port) throws IOException {
062        LOG.debug("creating new socket with " + timeout + "ms timeout");
063
064        Socket socket = new Socket();
065        socket.connect(new InetSocketAddress(host, port), timeout);
066
067        return socket;
068    }
069
070    /** {@inheritDoc} */
071    @Override
072    public final ServerSocket createServerSocket(final int port) throws IOException {
073        return new ServerSocket(port);
074    }
075}
