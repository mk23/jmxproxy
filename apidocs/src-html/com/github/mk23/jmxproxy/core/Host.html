001package com.github.mk23.jmxproxy.core;
002
003import com.fasterxml.jackson.core.JsonGenerator;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.JsonSerializable;
006import com.fasterxml.jackson.databind.SerializerProvider;
007import com.fasterxml.jackson.databind.jsontype.TypeSerializer;
008
009import java.io.IOException;
010
011import java.util.HashMap;
012import java.util.Map;
013import java.util.Set;
014
015/**
016 * <p>JMX Host tracker and serializer.</p>
017 *
018 * Maintains a map of JMX {@link MBean} names and their values. Implements
019 * JsonSerializable interface to convert the stored map into JSON.
020 *
021 * @see <a href="https://fasterxml.github.io/jackson-databind/javadoc/2.6/com/fasterxml/jackson/databind/JsonSerializable.html">com.fasterxml.jackson.databind.JsonSerializable</a>
022 *
023 * @since   2018-02-08
024 * @author  mk23
025 * @version 3.4.0
026 */
027public class Host implements JsonSerializable {
028    private final Map<String, MBean> mbeans;
029    private final ThreadLocal<Integer> limit = new ThreadLocal<Integer>() {
030        @Override protected Integer initialValue() {
031            return -1;
032        }
033    };
034
035    /**
036     * <p>Default constructor.</p>
037     *
038     * Creates a map of {@link MBean} name to associated values.
039     */
040    public Host() {
041        mbeans = new HashMap<String, MBean>();
042    }
043
044    /**
045     * <p>Inserts a new MBean name to value association.</p>
046     *
047     * Creates a new {@link MBean} object and inserts into the map store
048     * associating it to the specified mbean name. Uses default attribute
049     * {@link com.github.mk23.jmxproxy.util.History} size configuration.
050     *
051     * @param mbeanName name of the {@link MBean} used as the map key.
052     *
053     * @return the newly created {@link MBean} object.
054     */
055    public final MBean addMBean(final String mbeanName) {
056        if (!mbeans.containsKey(mbeanName)) {
057            MBean mbean = new MBean();
058            mbeans.put(mbeanName, mbean);
059        }
060
061        return mbeans.get(mbeanName);
062    }
063
064    /**
065     * <p>Inserts a new MBean name to value association.</p>
066     *
067     * Creates a new {@link MBean} object and inserts into the map store
068     * associating it to the specified mbean name. Uses custom attribute
069     * {@link com.github.mk23.jmxproxy.util.History} size configuration.
070     *
071     * @param mbeanName name of the {@link MBean} used as the map key.
072     * @param historySize number of items to preserve in the associated
073     * {@link com.github.mk23.jmxproxy.util.History}.
074     *
075     * @return the newly created {@link MBean} object.
076     */
077    public final MBean addMBean(final String mbeanName, final int historySize) {
078        if (!mbeans.containsKey(mbeanName)) {
079            MBean mbean = new MBean(historySize);
080            mbeans.put(mbeanName, mbean);
081        }
082
083        return mbeans.get(mbeanName);
084    }
085
086    /**
087     * <p>Deletes an mbean association.</p>
088     *
089     * Removes an associated {@link MBean} from the map store for this host.
090     *
091     * @param mbeanName name of the {@link MBean} to remove.
092     */
093    public final void removeMBean(final String mbeanName) {
094        mbeans.remove(mbeanName);
095    }
096
097    /**
098     * <p>Sets the thread local history request limit.</p>
099     *
100     * Set the thread local limit for all {@link com.github.mk23.jmxproxy.util.History} when
101     * serializing to JSON. Because this method returns its object, requesting serialization
102     * can be done * with a single statement.
103     *
104     * <p>For example:</p>
105     *
106     * <code>return Response.ok(host.setLimit(5)).build();</code>
107     *
108     * @see <a href="http://docs.oracle.com/javaee/7/api/javax/ws/rs/core/Response.html">javax.ws.rs.core.Response</a>
109     *
110     * @param bound the number of items to retreive from {@link com.github.mk23.jmxproxy.util.History}
111     *     for this thread.
112     *
113     * @return this host object for chaining calls.
114     */
115    public final Host setLimit(final Integer bound) {
116        limit.set(bound);
117        return this;
118    }
119
120    /**
121     * <p>Getter for mbean names.</p>
122     *
123     * Extracts and returns the unique {@link Set} of all currently stored {@link MBean} names.
124     *
125     * @return {@link Set} of {@link MBean} name {@link String}s.
126     */
127    public final Set<String> getMBeans() {
128        return mbeans.keySet();
129    }
130
131    /**
132     * <p>Getter for specific mbean.</p>
133     *
134     * Fetches the specified {@link MBean} from the map store.
135     *
136     * @param mbean name of the {@link MBean} to look up in the map store.
137     *
138     * @return {@link MBean} object if found, null otherwise.
139     */
140    public final MBean getMBean(final String mbean) {
141        return mbeans.get(mbean);
142    }
143
144    /** {@inheritDoc} */
145    public final void serialize(
146        final JsonGenerator jgen,
147        final SerializerProvider sp
148    ) throws IOException, JsonProcessingException {
149        serializeWithType(jgen, sp, null);
150    }
151
152    /** {@inheritDoc} */
153    public final void serializeWithType(
154        final JsonGenerator jgen,
155        final SerializerProvider sp,
156        final TypeSerializer ts
157    ) throws IOException, JsonProcessingException {
158        buildJson(jgen);
159    }
160
161    private void buildJson(final JsonGenerator jgen) throws IOException, JsonProcessingException {
162        int bound = limit.get();
163
164        jgen.writeStartObject();
165        for (Map.Entry<String, MBean> mbeanEntry : mbeans.entrySet()) {
166            jgen.writeObjectField(mbeanEntry.getKey(), mbeanEntry.getValue().setLimit(bound));
167        }
168        jgen.writeEndObject();
169    }
170}
