001package com.github.mk23.jmxproxy.core;
002
003import com.github.mk23.jmxproxy.util.History;
004
005import com.fasterxml.jackson.core.JsonGenerator;
006import com.fasterxml.jackson.core.JsonProcessingException;
007import com.fasterxml.jackson.databind.JsonSerializable;
008import com.fasterxml.jackson.databind.SerializerProvider;
009import com.fasterxml.jackson.databind.jsontype.TypeSerializer;
010
011import java.io.IOException;
012
013import java.util.Collections;
014import java.util.HashMap;
015import java.util.List;
016import java.util.Map;
017import java.util.Set;
018
019/**
020 * <p>JMX MBean tracker and serializer.</p>
021 *
022 * Maintains a map of JMX {@link Attribute} names and a {@link History} of
023 * their values.  Implements JsonSerializable interface to convert the
024 * stored map into JSON.
025 *
026 * @see <a href="https://fasterxml.github.io/jackson-databind/javadoc/2.6/com/fasterxml/jackson/databind/JsonSerializable.html">com.fasterxml.jackson.databind.JsonSerializable</a>
027 *
028 * @since   2018-02-08
029 * @author  mk23
030 * @version 3.4.0
031 */
032public class MBean implements JsonSerializable {
033    private final int historySize;
034
035    private final Map<String, History<Attribute>> attributes;
036    private final ThreadLocal<Integer> limit = new ThreadLocal<Integer>() {
037        @Override protected Integer initialValue() {
038            return -1;
039        }
040    };
041
042    /**
043     * <p>Default constructor.</p>
044     *
045     * Creates a map of {@link Attribute} name to {@link History} of associated values.
046     */
047    public MBean() {
048        this.historySize = 1;
049        attributes = new HashMap<String, History<Attribute>>();
050    }
051
052    /**
053     * <p>Default constructor.</p>
054     *
055     * Creates a map of {@link Attribute} name to {@link History} of associated values.
056     *
057     * @param historySize number of items to preserve in the associated {@link History}.
058     */
059    public MBean(final int historySize) {
060        this.historySize = historySize;
061        attributes = new HashMap<String, History<Attribute>>();
062    }
063
064    /**
065     * <p>Inserts a new Attribute name to History association.</p>
066     *
067     * Optionally creates a new {@link History} object and inserts into the map store associating
068     * it to the specified attribute name. Appends the specified {@link Attribute} value to the
069     * {@link History}.
070     *
071     * @param attributeName name of the {@link Attribute} used as the map key.
072     * @param attributeObject value of the {@link Attribute} added to history.
073     */
074    public final void addAttribute(final String attributeName, final Object attributeObject) {
075        if (!attributes.containsKey(attributeName)) {
076            attributes.put(attributeName, new History<Attribute>(historySize));
077        }
078
079        attributes.get(attributeName).add(new Attribute(attributeObject));
080    }
081
082    /**
083     * <p>Sets the thread local history request limit.</p>
084     *
085     * Set the thread local limit for all {@link History} when serializing to JSON.
086     * Because this method returns its object, requesting serialization can be done
087     * with a single statement.
088     *
089     * <p>For example</p>
090     *
091     * <code>return Response.ok(mbean.setLimit(5)).build();</code>
092     *
093     * @see <a href="http://docs.oracle.com/javaee/7/api/javax/ws/rs/core/Response.html">javax.ws.rs.core.Response</a>
094     *
095     * @param bound the number of items to retreive from {@link History} for this thread.
096     *
097     * @return this mbean object for chaining calls.
098     */
099    public final MBean setLimit(final Integer bound) {
100        limit.set(bound);
101        return this;
102    }
103
104    /**
105     * <p>Tester for attribute name.</p>
106     *
107     * Checks for requested {@link Attribute} name in the map store.
108     *
109     * @param attribute name of the {@link Attribute} to look up in the map store.
110     *
111     * @return true if the {@link Attribute} exists in the map store, false otherwise.
112     */
113    public final boolean hasAttribute(final String attribute) {
114        return attributes.containsKey(attribute);
115    }
116
117    /**
118     * <p>Getter for attribute names.</p>
119     *
120     * Extracts and returns the unique {@link Set} of all currently stored {@link Attribute} names.
121     *
122     * @return {@link Set} of {@link Attribute} name {@link String}s.
123     */
124    public final Set<String> getAttributes() {
125        return attributes.keySet();
126    }
127
128    /**
129     * <p>Getter for most recent attribute.</p>
130     *
131     * Fetches the most recent {@link Attribute} value for the specified name from the
132     * associated {@link History} in the map store.
133     *
134     * @param attribute name of the {@link Attribute} to look up in the map store.
135     *
136     * @return latest {@link Attribute} object from {@link History} if found, null otherwise.
137     */
138    public final Attribute getAttribute(final String attribute) {
139        if (hasAttribute(attribute)) {
140            return attributes.get(attribute).getLast();
141        }
142
143        return null;
144    }
145
146    /**
147     * <p>Getter for a subset of historical attribute values.</p>
148     *
149     * Fetches an array, limited by the requested bound, of most recent {@link Attribute} values
150     * for the specified name from the associated {@link History} in the map store.
151     *
152     * @param attribute name of the {@link Attribute} to look up in the map store.
153     * @param bound size of the resulting array or full history if this exceeds capacity or less than 1.
154     *
155     * @return {@link List} of the latest {@link Attribute} objects from {@link History} if found, empty otherwise.
156     */
157    public final List<Attribute> getAttributes(final String attribute, final int bound) {
158        if (hasAttribute(attribute)) {
159            return attributes.get(attribute).get(bound);
160        }
161
162        return Collections.emptyList();
163    }
164
165    /** {@inheritDoc} */
166    @Override
167    public final void serialize(
168        final JsonGenerator jgen,
169        final SerializerProvider sp
170    ) throws IOException, JsonProcessingException {
171        serializeWithType(jgen, sp, null);
172    }
173
174    /** {@inheritDoc} */
175    @Override
176    public final void serializeWithType(
177        final JsonGenerator jgen,
178        final SerializerProvider sp,
179        final TypeSerializer ts
180    ) throws IOException, JsonProcessingException {
181        buildJson(jgen);
182    }
183
184    private void buildJson(final JsonGenerator jgen) throws IOException, JsonProcessingException {
185        int bound = limit.get();
186
187        jgen.writeStartObject();
188        for (Map.Entry<String, History<Attribute>> attributeEntry : attributes.entrySet()) {
189            if (bound < 0) {
190                jgen.writeObjectField(attributeEntry.getKey(), attributeEntry.getValue().getLast());
191            } else {
192                jgen.writeObjectField(attributeEntry.getKey(), attributeEntry.getValue().get(bound));
193            }
194        }
195        jgen.writeEndObject();
196    }
197}
